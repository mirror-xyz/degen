---
title: SkeletonGroup
description: Groups and manages loading state for multiple skeletons.
---

```tsx
import { SkeletonGroup } from 'degen'
```

```tsx expand=true live=true
<SkeletonGroup loading>
  <Skeleton>Hello World</Skeleton>
</SkeletonGroup>
```

## Wrap content

```tsx live=true
<SkeletonGroup loading={getState('loading')}>
  <Stack direction="vertical">
    <Skeleton>
      <Heading level="1">Foo Bar Baz</Heading>
    </Skeleton>
    <Skeleton>
      <Text>The quick brown fox jumped over the lazy dogs.</Text>
    </Skeleton>

    <Button size="medium" onClick={() => toggleState('loading')}>
      Toggle Skeletom
    </Button>
  </Stack>
</SkeletonGroup>
```
