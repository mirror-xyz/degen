## States

```tsx live=true
<Stack direction="horizontal" wrap>
  <Button loading>Loading</Button>
  <Button disabled>Disabled</Button>
  <Button
    disabled={getState('loading')}
    loading={getState('loading')}
    onClick={() => {
      toggleState('loading')
      setTimeout(() => toggleState('loading'), 1000)
    }}
  >
    Click Me
  </Button>
</Stack>
```

## Tones

When `variant` is set to `highlight` or `primary`, you can `tone` to change the button accent color.

```tsx live=true
<Stack space="8">
  <Stack>
    <Text variant="label">Highlight</Text>
    <Stack direction="horizontal" wrap>
      <Button tone="accent" variant="highlight">
        Accent
      </Button>
      <Button tone="blue" variant="highlight">
        Blue
      </Button>
      <Button tone="green" variant="highlight">
        Green
      </Button>
      <Button tone="red" variant="highlight">
        Red
      </Button>
    </Stack>
  </Stack>

  <Stack>
    <Text variant="label">Primary</Text>
    <Stack direction="horizontal" wrap>
      <Button tone="accent" variant="primary">
        Accent
      </Button>
      <Button tone="blue" variant="primary">
        Blue
      </Button>
      <Button tone="green" variant="primary">
        Green
      </Button>
      <Button tone="red" variant="primary">
        Red
      </Button>
    </Stack>
  </Stack>
</Stack>
```

## Affixes

Optional `center` prop reserves space for affixes and loading spinner.

```tsx live=true
<Stack space="8">
  <Stack>
    <Text variant="label">Default</Text>
    <Stack direction="horizontal" wrap>
      <Button prefix={<IconPlusSmall />}>Add Entry</Button>
      <Button suffix={<IconPlusSmall />}>Add Entry</Button>
      <Button prefix={<IconPlusSmall />} suffix={<IconPlusSmall />}>
        Add Entry
      </Button>
      <Button loading prefix={<IconPlusSmall />}>
        Add Entry
      </Button>
    </Stack>
  </Stack>

  <Stack>
    <Text variant="label">Center</Text>
    <Stack direction="horizontal" wrap>
      <Button center prefix={<IconPlusSmall />}>
        Add Entry
      </Button>
      <Button center suffix={<IconPlusSmall />}>
        Add Entry
      </Button>
      <Button center prefix={<IconPlusSmall />} suffix={<IconPlusSmall />}>
        Add Entry
      </Button>
      <Button center loading prefix={<IconPlusSmall />}>
        Add Entry
      </Button>
    </Stack>
  </Stack>
</Stack>
```
